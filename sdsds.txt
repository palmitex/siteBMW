    <script>
        document.getElementById('contactForm').addEventListener('submit', function(event) {
            var cpfInput = document.getElementById('cpf').value;
            var cpfPattern = /^\d{3}\.\d{3}\.\d{3}-\d{2}$/;
            
            if (!cpfPattern.test(cpfInput)) {
                alert('Por favor, insira um CPF válido no formato 123.456.789-00');
                event.preventDefault(); // Impede o envio do formulário se o CPF estiver incorreto
            }
        });
